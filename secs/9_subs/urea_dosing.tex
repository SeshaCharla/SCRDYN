\subsubsection{Urea Dosing Dynamics}
We have,
\begin{align*}
        \con{NH_3}^{in}(k) &= \nu_u \underbrace{u_{inj}(k)}_{u_2} - \nu_T T - \nu_F F
\end{align*}
\begin{align}
        \implies \con{NH_3}^{in}(k)  &= \pmb \phi_{ur}(k) ^T \pmb \theta_{ur}
\end{align}
Where,
\begin{align*}
        \pmb \phi_{ur} &= \bm{u_{inj} & -T & -F}\\
        \pmb \theta_{ur} &= \bm{\nu_u & \nu_T & \nu_F}
\end{align*}
